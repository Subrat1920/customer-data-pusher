SELECT 
    c1.column_name AS churn_data_columns,
    c2.column_name AS customer_data_columns
FROM 
    (SELECT column_name, ROW_NUMBER() OVER (ORDER BY ordinal_position) AS rn
     FROM information_schema.columns 
     WHERE table_name = 'churn_data') c1
FULL OUTER JOIN
    (SELECT column_name, ROW_NUMBER() OVER (ORDER BY ordinal_position) AS rn
     FROM information_schema.columns
     WHERE table_name = 'customer_data') c2
ON c1.rn = c2.rn
ORDER BY COALESCE(c1.rn, c2.rn);
